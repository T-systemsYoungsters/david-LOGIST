import random
heads=0
tails=0
for i in range(50):
    match random.randint(0,1):
        case 0:
            print("heads")
            heads+=1
        case 1:
            print("tails")
            tails+=1
print("The number of total heads is {}, the number of total tails is {}.".format(heads, tails))
